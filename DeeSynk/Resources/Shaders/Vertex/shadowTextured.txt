#version 450 core

layout (location = 0) in vec4 position;
layout (location = 1) in vec2 texCoord;
layout (location = 3) uniform int lightNum;
layout (location = 13) uniform mat4 modelMat;

layout(binding = 3) uniform SpotLight
{
	mat4 spotLightView;
	vec4 spotLightLocation;
	vec4 spotLightLookAt;
	vec4 spotLightColor;
	vec4 spotLightFOV;
} spotLights[4];

void main(void)
{
 gl_Position = spotLights[lightNum].spotLightView * modelMat * (position);
}
