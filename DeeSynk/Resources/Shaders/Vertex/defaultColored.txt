layout (location = 1) in vec4 color;

out vec4 vs_color;

{
	mat4 projection;  //model_view
	vec4 camLocation; //cameraLoc
} camera;

{
 vs_color = vec4(1, 1, 1, 1);
 vec4 pos = position;
 pos.w = 1;
 gl_Position = camera.projection * model_view * pos;
}
