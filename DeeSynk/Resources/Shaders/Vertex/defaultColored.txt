#version 450 core

layout (location = 0) in vec4 position;
layout (location = 1) in vec4 color;
layout (location = 2) uniform mat4 ortho;
layout (location = 3) uniform mat4 rot;
layout (location = 4) uniform mat4 tra;
layout (location = 5) uniform mat4 sca;
out vec4 vs_color;

void main(void)
{
 vs_color = color;
 gl_Position = ortho * tra * rot * sca * position;
}
