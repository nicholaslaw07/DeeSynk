#version 450 core

layout (location = 0) in vec4 position;
layout (location = 1) in vec2 texCoord;

layout (location = 10) uniform mat4 viewProjection;
uniform mat4 model[100];

out vec2 vs_texCoord;

void main(void)
{
 vs_texCoord = texCoord;
 gl_Position = model[gl_InstanceID] * viewProjection * position;
}
