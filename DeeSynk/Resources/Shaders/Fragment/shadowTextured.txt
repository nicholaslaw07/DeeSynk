#version 450 core

in vec2 vs_texCoord;
layout(binding=0) uniform sampler2D tex;
layout(binding=1) uniform sampler2D shadow;
layout(binding=2) uniform sampler2D shadowTest;
out vec4 color;

void main(void)
{
	float Depth = texture(shadow, vs_texCoord).x;
	Depth = 1.0 - (1.0 - Depth) * 25.0;
	vec4 c = vec4(Depth);
	c.a = 1;
	color = c;
}
